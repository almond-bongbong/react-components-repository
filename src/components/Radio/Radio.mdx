## Basic usage

<Playground>
  {() => {
    class Test extends React.Component {
      constructor (props) {
        super(props)

        this.state = { checkedIdx: 0 }
      }

      render() {
        return (
          <>
            <Radio id="radio1-1" label="라디오1" checked={this.state.checkedIdx === 0} onChange={e => e.target.checked && this.setState({ checkedIdx: 0 })} />
            <Radio id="radio1-2" label="라디오2" checked={this.state.checkedIdx === 1} onChange={e => e.target.checked && this.setState({ checkedIdx: 1 })} />
          </>
        )
      }
    }
    return (
      <Test />
    )
  }}
</Playground>
