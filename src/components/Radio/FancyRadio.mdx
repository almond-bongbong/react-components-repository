## Basic usage

<Playground>
  {() => {
    class Test extends React.Component {
      constructor (props) {
        super(props)

        this.state = { checkedIdx: 0 }
      }

      render() {
        return (
          <SelectGroup>
            <FancyRadio id="radio1-1" label="male" checked={this.state.checkedIdx === 0} onChange={e => e.target.checked && this.setState({ checkedIdx: 0 })} />
            <FancyRadio id="radio1-2" label="female" checked={this.state.checkedIdx === 1} onChange={e => e.target.checked && this.setState({ checkedIdx: 1 })} />
          </SelectGroup>
        )
      }
    }
    return (
      <Test />
    )
  }}
</Playground>
