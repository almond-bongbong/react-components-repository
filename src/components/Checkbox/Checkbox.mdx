## Basic usage

<Playground>
  {() => {
    class Test extends React.Component {
      constructor (props) {
        super(props)

        this.state = { checked: false }
      }

      render() {
        return (
          <>
            <Checkbox id="chk1-1" label="체크박스" checked={this.state.checked} onChange={e => this.setState({ checked: e.target.checked })} />
          </>
        )
      }
    }
    return (
      <Test />
    )
  }}
</Playground>
